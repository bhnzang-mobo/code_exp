#include <stdio.h>
void merge_sort(int * arr, int low, int high){
    if(low<high){
        int left= low;
        int tmp[high-low+1];
        int q= (high+low-1)/2;
        merge_sort(arr,low,q);
        merge_sort(arr,q+1,high);
        int k = q+1;
        int i = 0;
        while(low<=q && k<=high){
            tmp[i++]=(arr[low]<=arr[k])?arr[low++]:arr[k++];
        }
        while(low<=q){
            tmp[i++]=arr[low++];
        }
        while (k<=high){
            tmp[i++]=arr[k++];
        }
        for(int a = left; a<=high ; a++){
            arr[a]=tmp[a-left];
        }
    }
}
int main() {
    int arr[10]={1,2,3,6,8,7,5,43,45,13};
    merge_sort(arr,0,9);
    return 0;
}
